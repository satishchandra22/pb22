const Calculator = (A, B, C) => 
{
	
  let out; 
 switch(A) {
      case '+':
      out=B+C;
      break;

      case '-':
      out=B-C;
      break;

      case '*':
      out=B*C;
      break;

      case '/':
      out=B/C;
      break;
}
 return out;
}

const getValue = (a) => {

    return 'PrepBytes'
  }else if (a==='Z' || a==='z'){

  }else if (a==='E' || a==='e'){

};


const Max_out_of_three = (A,B,C) => {
  if (A===B && A===C){
    return -1
  } else if (B>=A && B>=C){
    return B
  } else if (A>=B && A>=C){
    return A
  } else{
    return C
  }

};


const findSndSmallest = (x,y,z) => {
   if((x < y && y < z) || (z < y && y < x)){
     return y;
   }else if((y < x && x < z) || (z < x && x < y)){
     return x;
   }else{
     return z;
   } 
};
